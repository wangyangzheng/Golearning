##### 第一种模版

当目标值在mid的左边时,区间变成`[mid, right]`,否者变成`[left, mid - 1]`
```go
func search(nums []int, target int) int {
    left, right := 0, len(nums) - 1
    for left < right {
        //注意这里的加1操作
        mid := (left + right + 1) >> 1
        if nums[mid] <= target {
            left = mid
        } else {
            right = mid - 1
        }
    }
    if nums[left] == target {
        return left
    } else {
        return -1
    }
}
```

#### 第二种模板

当目标值在mid的右边的时候,区间变成`[left, mid]`,否则编程`[mid + 1, right]`
```go
func search(nums []int, target int) int {
    left, right := 0, len(nums) - 1
    for left < right {
        mid := (left + right) >> 1;
        if nums[mid] >= target {
            right = mid
        } else {
            left = mid + 1
        }
    }
    if nums[left] == target {
        return left
    } else {
        return -1
    }
}
```

两种模版的结合

package main

import "fmt"

func main(){
    n, m := 0, 0
    fmt.Scan(&n, &m)
    
    q := make([]int, 1000010)
    for i := 0; i < n; i++ {
        fmt.Scan(&q[i])
    }
    
    for ; m > 0; m-- {
        a := 0
        fmt.Scan(&a)
        left, right := 0, n - 1
        for left < right {
            mid := (left + right)>>1
            if q[mid] >= a {
                right = mid
            } else {
                left = mid + 1
            }
        }
        if q[left] == a {
            fmt.Print(left, " ")
            left, right := 0, n - 1
            for left < right {
                mid := (left + right + 1) >> 1
                if q[mid] <= a {
                    left = mid
                } else {
                    right = mid - 1
                }
            }
            fmt.Println(right)
        } else {
            fmt.Println("-1 -1")
        }
    }
}
