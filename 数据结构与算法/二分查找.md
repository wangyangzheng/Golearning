##### 第一种模版

当目标值在mid的左边时
```go
func search(nums []int, target int) int {
    left, right := 0, len(nums) - 1
    for left < right {
        //注意这里的加1操作
        mid := (left + right + 1) >> 1
        if nums[mid] <= target {
            left = mid
        } else {
            right = mid - 1
        }
    }
    if nums[left] == target {
        return left
    } else {
        return -1
    }
}
```

#### 第二种模板

当目标值在mid的右边的时候
```go
func search(nums []int, target int) int {
    left, right := 0, len(nums) - 1
    for left < right {
        mid := (left + right) >> 1;
        if nums[mid] >= target {
            right = mid
        } else {
            left = mid + 1
        }
    }
    if nums[left] == target {
        return left
    } else {
        return -1
    }
}
```
