```go
package main

import "fmt"

var (
    num = make([]int, 100010)
    n int
)

func quick_sort(q [] int, left, right int) {
    if left >= right {
        return 
    }
    
    mid := q[(left + right) >> 1]
    i, j := left - 1, right + 1
    
    for i < j {
       for {
           i++
           if q[i] >= mid {
               break
           }
       }
       for {
           j--
           if q[j] <= mid {
               break
           }
       }
       if i < j {
           q[i], q[j] = q[j], q[i]
       }
    }
    quick_sort(q, left, j)
    quick_sort(q, j + 1, right)
}

func main(){
    
    fmt.Scan(&n)
    for i := 0; i < n; i++ {
        fmt.Scan(&num[i])
    }
    quick_sort(num, 0, n - 1)
    for i := 0; i < n; i++ {
        fmt.Printf("%d ", num[i])
    }
}
```
